import pandas as pd
import matplotlib.pyplot as plt

file_path = '/Users/irasamant/Downloads/tech_companies_salary (1).csv'
data = pd.read_csv(file_path)

def filter_race_and_title(data, race):
    return data.loc[
        (data['title'].str.strip() == 'Software Engineer') &
        (data['Race'].str.strip() == race) & 
        (data['gender'].notna()) & 
        (data['totalyearlycompensation'].notna())
    ].copy()

white_software_df = filter_race_and_title(data, 'White')
asian_software_df = filter_race_and_title(data, 'Asian')
black_software_df = filter_race_and_title(data, 'Black')

white_software_df['Race'] = 'White'
asian_software_df['Race'] = 'Asian'
black_software_df['Race'] = 'Black'

combined_df = pd.concat([white_software_df, asian_software_df, black_software_df])

plt.figure(figsize=(12, 6))
combined_df.boxplot(
    column='totalyearlycompensation', 
    by=['Race', 'gender'], 
    grid=False, 
    patch_artist=True
)

plt.title('Salary Distribution by Gender for White, Asian, and Black Software Engineers', fontsize=16)
plt.suptitle('')  # Suppress the default suptitle
plt.xlabel('Race and Gender', fontsize=14)
plt.ylabel('Total Yearly Compensation ($)', fontsize=14)
plt.xticks(rotation=45, fontsize=12)
plt.yticks(fontsize=12)

plt.show()

import pandas as pd
import matplotlib.pyplot as plt

file_path = '/Users/irasamant/Downloads/tech_companies_salary (1).csv'
data = pd.read_csv(file_path)


def filter_race_and_title(data, race):
    return data.loc[
        (data['title'].str.strip() == 'Software Engineer') &
        (data['Race'].str.strip() == race) & 
        (data['gender'].notna()) & 
        (data['totalyearlycompensation'].notna())
    ].copy()


races = ['White', 'Asian', 'Black']


for race in races:
    race_df = filter_race_and_title(data, race)
    
    
    plt.figure(figsize=(8, 6))

        column='totalyearlycompensation', 
        by='gender', 
        grid=False, 
        patch_artist=True

    plt.show()
