import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

dataset = pd.read_csv(r"C:\Users\pcadu\Downloads\tech_companies_salary.csv")

job_title_filter = 'Software Engineer'  
quantitative_var = 'totalyearlycompensation'  
binary_var = 'Race_White' 

filtered_dataset = dataset[dataset['title'] == job_title_filter]

plt.figure(figsize=(8, 6))

sns.regplot(data=filtered_dataset, x=binary_var, y=quantitative_var, logistic=True) 

def format_ticks(value, _):
    return f"${value / 1e5:.0f}0K" 

plt.gca().yaxis.set_major_formatter(FuncFormatter(format_ticks))


plt.title(f"Regression of {quantitative_var} by {binary_var}\nFiltered by Job Title = {job_title_filter}")
plt.xlabel(binary_var)
plt.ylabel(quantitative_var)


plt.tight_layout()


plt.show()

import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

dataset = pd.read_csv(r"C:\Users\pcadu\Downloads\tech_companies_salary.csv")

job_title_filter = 'Software Engineer'  
quantitative_var = 'totalyearlycompensation'  
binary_var1 = 'Race_White'  
binary_var2 = 'Race_Black'  

filtered_dataset = dataset[dataset['title'] == job_title_filter]

plt.figure(figsize=(8, 6))

sns.regplot(data=filtered_dataset, x=binary_var1, y=quantitative_var, logistic=True, label=f"{binary_var1} vs {quantitative_var}")
sns.regplot(data=filtered_dataset, x=binary_var2, y=quantitative_var, logistic=True, label=f"{binary_var2} vs {quantitative_var}")

def format_ticks(value, _):
    return f"${value / 1e5:.0f}0K"  

plt.gca().yaxis.set_major_formatter(FuncFormatter(format_ticks))

plt.title(f"Logistic Regression of {quantitative_var} by {binary_var1} and {binary_var2}\nFiltered by Job Title = {job_title_filter}")
plt.xlabel(f"{binary_var1} and {binary_var2}")
plt.ylabel(quantitative_var)

plt.tight_layout()

plt.show()

import pandas as pd

import matplotlib.pyplot as plt
from matplotlib.ticker import FuncFormatter

dataset = pd.read_csv(r"C:\Users\pcadu\Downloads\tech_companies_salary.csv")

job_title_filter = 'Software Engineer'  
quantitative_var = 'totalyearlycompensation' 
binary_var1 = 'gender'  

filtered_dataset = dataset[dataset['title'] == job_title_filter]

filtered_dataset[binary_var1] = filtered_dataset[binary_var1].map({'Female': 0, 'Male': 1})

plt.figure(figsize=(8, 6))

sns.lmplot(data=filtered_dataset, x=binary_var1, y=quantitative_var, logistic=False, scatter_kws={'s': 10}, line_kws={'color': 'blue'})

    return f"${value / 1e5:.0f}0K" 
    
plt.gca().yaxis.set_major_formatter(FuncFormatter(format_ticks))

plt.title(f"Linear Regression of {quantitative_var} by {binary_var1}\nFiltered by Job Title = {job_title_filter}")
plt.xlabel(f"{binary_var1} (Male vs Female)")
plt.ylabel(quantitative_var)

plt.tight_layout()
plt.show()
